#include<iostream>
using namespace std;
int h[100];
int main(){
int T,N,P;
cin>>T;
while(T--){
cin>>N>>P;
int day[3651]={};
for(int i=0;i<P;i++){
cin>>h[i];
for(int j=h[i];j<=N;j+=h[i])
day[j]=1;
}
int count=0;
for(int j=1;j<=N;j++){
if(j%7==6 || j%7==0)continue;
if(day[j]==1)count++;
}
cout<<count<<endl;
}
}
